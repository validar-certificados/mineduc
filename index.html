// --- CONFIGURACIÓN DEL SERVIDOR --- // 1. Nombre EXACTO de la carpeta en Drive que funciona como 'base validadora' var NOMBRE_CARPETA_BASE = "base validadora"; function doGet(e) { // Maneja la solicitud de página (puedes ignorar esto si solo usas el HTML externo) return HtmlService.createHtmlOutputFromFile('Index'); } function doPost(e) { // Maneja la búsqueda y validación try { var data = JSON.parse(e.postData.contents); var codigoVerificacion = data.code; if (!codigoVerificacion) { return response({ success: false, message: "No se proporcionó código." }); } // 1. Buscar la carpeta 'base validadora' var carpetas = DriveApp.getFoldersByName(NOMBRE_CARPETA_BASE); if (!carpetas.hasNext()) { return response({ success: false, message: "Carpeta 'base validadora' no encontrada en Drive." }); } var carpetaBase = carpetas.next(); // 2. Buscar el archivo por nombre exacto dentro de esa carpeta // Asumimos que el código ingresado coincide con el nombre del archivo (con o sin extensión) var query = "name = '" + codigoVerificacion + "' or name = '" + codigoVerificacion + ".pdf'"; var archivos = carpetaBase.searchFiles(query); if (archivos.hasNext()) { var archivoValidado = archivos.next(); // 3. Obtener el archivo como base64 para embedirlo en el HTML var blob = archivoValidado.getBlob(); var contentType = blob.getContentType(); var base64Data = Utilities.base64Encode(blob.getBytes()); var fileUrl = "data:" + contentType + ";base64," + base64Data; return response({ success: true, message: "Documento válido y vigente.", fileName: archivoValidado.getName(), fileType: contentType, fileContent: fileUrl // URL base64 lista para el iframe/object }); } else { return response({ success: false, message: "Código de verificación no encontrado o documento no válido." }); } } catch (error) { return response({ success: false, message: "Error interno del servidor: " + error.toString() }); } } // Función auxiliar para responder en formato JSON function response(obj) { return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON); }
